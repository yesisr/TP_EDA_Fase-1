203Antena *adicionarEfeitoNefasto(Antena *lista)
204{
205 Antena *atual = lista;
206
207 while (atual)
208 {
209 Antena *proximo = atual->next;
210
211 while (proximo)
212 {
213 // precorre a lista até encontrar duas antenas com a mesma frequencia
214 if (atual->freq == proximo->freq)
215 {
216 // distancia de coordenadas entre a proxima antena e atual
217 // para calcular o efeito nefasto
218 // exmpl: 2,2 - 1,1 = 1,1
219 int diferencaX = proximo->x - atual->x;
220 int diferencaY = proximo->y - atual->y;
221
222 // coordenadas do primeiro efeito nefasto
223 // exmpl: 1,1 - 1,1 = 0,0
224 int efeitoX1 = atual->x - diferencaX;
225 int efeitoY1 = atual->y - diferencaY;
226
227 // coordenadas do segundo efeito nefasto
228 // exmpl: 2,2 + 1,1 = 3,3
229 int efeitoX2 = proximo->x + diferencaX;
230 int efeitoY2 = proximo->y + diferencaY;
231
232 // verifica se as coordenadas são possiveis e se existe antena nas coordenadas
233 if (efeitoX1 >= 1 && efeitoX1 <= max_x &&
234 efeitoY1 >= 1 && efeitoY1 <= max_y &&
235 !existeAntena(lista, efeitoX1, efeitoY1))
236 {
237 // cria efeito antes da primeira antena
238 lista = inserirNefasto(lista, criarAntena(efeitoX1, efeitoY1, '#'));
239 }
240 // verifica se as coordenadas são possiveis e se existe antena nas coordenadas
241 if (efeitoX2 >= 1 && efeitoX2 <= max_x &&
242 efeitoY2 >= 1 && efeitoY2 <= max_y &&
243 !existeAntena(lista, efeitoX2, efeitoY2))
244 {
245 // cria efeito depois da segunda antena
246 lista = inserirNefasto(lista, criarAntena(efeitoX2, efeitoY2, '#'));
247 }
248 }
249 proximo = proximo->next;
250 }
251 atual = atual->next;
252 }
253
254 return lista;
255}
